\section{Derivatives Used in the Bond-Pricing PDE}
\label{app:pde_derivatives}

Define for convenience
\[
\Phi(z,\tau):=A_z(\tau)-B_z(\tau)\,G(z,\tau),
\qquad\text{so that}\qquad
P(z,\tau)=e^{\Phi(z,\tau)}.
\]

\textbf{(i) $\tau$-derivative.}
By the chain rule,
\[
\partial_\tau P = (\partial_\tau \Phi)\,P.
\]
Moreover,
\[
\partial_\tau \Phi
=
\partial_\tau A_z(\tau) - \partial_\tau\!\big(B_z(\tau)\,G(z,\tau)\big)
=
\partial_\tau A_z(\tau)
-
(\partial_\tau B_z(\tau))\,G(z,\tau)
-
B_z(\tau)\,\partial_\tau G(z,\tau).
\]
Hence,
\[
\boxed{
\partial_\tau P
=
\Big(
\partial_\tau A_z
-
(\partial_\tau B_z)\,G
-
B_z\,\partial_\tau G
\Big)P.
}
\]

\textbf{(ii) Gradient in $z$.}
Following the state-wise derivation used in the main text, \(A_z(\tau)\) and \(B_z(\tau)\) are treated as maturity-dependent coefficient paths for a fixed state \(z\). Hence their \(z\)-derivatives are not included in this calculation.
The $z$-dependence of $\Phi$ comes only through $G(z,\tau)$:
\[
\nabla_z \Phi = -B_z(\tau)\,\nabla_z G(z,\tau).
\]
Thus,
\[
\boxed{
\nabla_z P
=
(\nabla_z \Phi)\,P
=
\big(-B_z\,\nabla_z G\big)P.
}
\] 

\textbf{(iii) Hessian in $z$.}
A standard identity for $P=e^{\Phi}$ is
\[
H_z P = P\big(H_z\Phi + (\nabla_z\Phi)(\nabla_z\Phi)^\top\big).
\]
We have
\[
H_z\Phi = -B_z(\tau)\,H_z G(z,\tau),
\qquad
(\nabla_z\Phi)(\nabla_z\Phi)^\top
=
B_z(\tau)^2\,\nabla_z G(z,\tau)\,\nabla_z G(z,\tau)^\top.
\]
Therefore,
\[
\boxed{
H_z P
=
P\Big(
-B_z\,H_z G
+
B_z^2\,\nabla_z G\,\nabla_z G^\top
\Big).
}
\] 



\section{Cholesky Parameterization of the Volatility Matrix}
\label{sec:cholesky_volatility}

To ensure that the instantaneous covariance matrix is positive semidefinite, we parameterize it through a lower-triangular Cholesky factor. Let
\[
\Sigma = L L^\top,
\]
where \(L\) is lower triangular. We further write
\[
\Sigma = D R D,
\qquad
D = \operatorname{diag}(\sigma_1,\dots,\sigma_d),
\]
where \(\sigma_i > 0\) are factor volatilities and \(R\) is the correlation matrix. Hence, if
\[
R = C C^\top,
\]
then
\[
L = D C.
\]

Below we derive the explicit form of \(L\) for dimensions \(d=2,3,4\).

\subsection{Two-dimensional case}

In dimension \(2\), the correlation matrix is
\[
R =
\begin{bmatrix}
1 & \rho_{12} \\
\rho_{12} & 1
\end{bmatrix}.
\]
We seek a lower-triangular Cholesky factor
\[
C =
\begin{bmatrix}
c_{11} & 0 \\
c_{21} & c_{22}
\end{bmatrix}
\]
such that
\[
R = C C^\top.
\]
Matching entries gives
\[
c_{11}^2 = 1,
\qquad
c_{11}c_{21} = \rho_{12},
\qquad
c_{21}^2 + c_{22}^2 = 1.
\]
Choosing the positive diagonal gives
\[
c_{11}=1,
\qquad
c_{21}=\rho_{12},
\qquad
c_{22}=\sqrt{1-\rho_{12}^2}.
\]
Therefore,
\[
C =
\begin{bmatrix}
1 & 0 \\
\rho_{12} & \sqrt{1-\rho_{12}^2}
\end{bmatrix}.
\]
Multiplying by \(D=\operatorname{diag}(\sigma_1,\sigma_2)\) yields
\[
L = D C =
\begin{bmatrix}
\sigma_1 & 0 \\
\rho_{12}\sigma_2 & \sigma_2\sqrt{1-\rho_{12}^2}
\end{bmatrix}.
\]

\subsection{Three-dimensional case}

In dimension \(3\), the correlation matrix is
\[
R =
\begin{bmatrix}
1 & \rho_{12} & \rho_{13} \\
\rho_{12} & 1 & \rho_{23} \\
\rho_{13} & \rho_{23} & 1
\end{bmatrix}.
\]
We write
\[
C =
\begin{bmatrix}
1 & 0 & 0 \\
c_{21} & c_{22} & 0 \\
c_{31} & c_{32} & c_{33}
\end{bmatrix}
\]
and impose \(R = C C^\top\). Matching entries row by row:

From the \((2,1)\)-entry,
\[
c_{21} = \rho_{12}.
\]
From the \((2,2)\)-entry,
\[
c_{21}^2 + c_{22}^2 = 1
\quad\Longrightarrow\quad
c_{22} = \sqrt{1-\rho_{12}^2}.
\]
From the \((3,1)\)-entry,
\[
c_{31} = \rho_{13}.
\]
From the \((3,2)\)-entry,
\[
c_{31}c_{21} + c_{32}c_{22} = \rho_{23}.
\]
Substituting \(c_{21}=\rho_{12}\), \(c_{22}=\sqrt{1-\rho_{12}^2}\), and \(c_{31}=\rho_{13}\), we obtain
\[
c_{32}
=
\frac{\rho_{23}-\rho_{12}\rho_{13}}{\sqrt{1-\rho_{12}^2}}.
\]
Finally, from the \((3,3)\)-entry,
\[
c_{31}^2 + c_{32}^2 + c_{33}^2 = 1,
\]
hence
\[
c_{33}
=
\sqrt{
1-\rho_{13}^2
-\frac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}.
\]
Thus,
\[
C =
\begin{bmatrix}
1 & 0 & 0 \\
\rho_{12} & \sqrt{1-\rho_{12}^2} & 0 \\
\rho_{13} &
\dfrac{\rho_{23}-\rho_{12}\rho_{13}}{\sqrt{1-\rho_{12}^2}} &
\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
\end{bmatrix}.
\]
Therefore,
\[
L = D C =
\begin{bmatrix}
\sigma_1 & 0 & 0 \\
\rho_{12}\sigma_2 & \sigma_2\sqrt{1-\rho_{12}^2} & 0 \\
\rho_{13}\sigma_3 &
\sigma_3\dfrac{\rho_{23}-\rho_{12}\rho_{13}}{\sqrt{1-\rho_{12}^2}} &
\sigma_3\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
\end{bmatrix}.
\]

\subsection{Four-dimensional case}

In dimension \(4\), the correlation matrix is
\[
R =
\begin{bmatrix}
1 & \rho_{12} & \rho_{13} & \rho_{14} \\
\rho_{12} & 1 & \rho_{23} & \rho_{24} \\
\rho_{13} & \rho_{23} & 1 & \rho_{34} \\
\rho_{14} & \rho_{24} & \rho_{34} & 1
\end{bmatrix}.
\]
We write
\[
C =
\begin{bmatrix}
1 & 0 & 0 & 0 \\
c_{21} & c_{22} & 0 & 0 \\
c_{31} & c_{32} & c_{33} & 0 \\
c_{41} & c_{42} & c_{43} & c_{44}
\end{bmatrix}
\]
and again impose \(R=C C^\top\).

The first three rows follow exactly as in the three-dimensional case:
\[
c_{21}=\rho_{12},
\qquad
c_{22}=\sqrt{1-\rho_{12}^2},
\]
\[
c_{31}=\rho_{13},
\qquad
c_{32}=\frac{\rho_{23}-\rho_{12}\rho_{13}}{\sqrt{1-\rho_{12}^2}},
\]
\[
c_{33}
=
\sqrt{
1-\rho_{13}^2
-\frac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}.
\]

For the fourth row, the \((4,1)\)-entry gives
\[
c_{41}=\rho_{14}.
\]
The \((4,2)\)-entry gives
\[
c_{41}c_{21}+c_{42}c_{22}=\rho_{24},
\]
hence
\[
c_{42}
=
\frac{\rho_{24}-\rho_{12}\rho_{14}}{\sqrt{1-\rho_{12}^2}}.
\]
The \((4,3)\)-entry gives
\[
c_{41}c_{31}+c_{42}c_{32}+c_{43}c_{33}=\rho_{34},
\]
so
\[
c_{43}
=
\frac{\rho_{34}-\rho_{13}\rho_{14}-c_{32}c_{42}}{c_{33}}.
\]
Substituting the expressions for \(c_{32}\) and \(c_{42}\),
\[
c_{43}
=
\frac{
\rho_{34}-\rho_{13}\rho_{14}
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})(\rho_{24}-\rho_{12}\rho_{14})}{1-\rho_{12}^2}
}{
\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
}.
\]
Finally, the \((4,4)\)-entry gives
\[
c_{41}^2+c_{42}^2+c_{43}^2+c_{44}^2=1,
\]
hence
\[
c_{44}
=
\sqrt{1-\rho_{14}^2-c_{42}^2-c_{43}^2}.
\]
Equivalently,
\[
c_{44}
=
\sqrt{
1-\rho_{14}^2
-\frac{(\rho_{24}-\rho_{12}\rho_{14})^2}{1-\rho_{12}^2}
-\left(
\frac{
\rho_{34}-\rho_{13}\rho_{14}
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})(\rho_{24}-\rho_{12}\rho_{14})}{1-\rho_{12}^2}
}{
\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
}
\right)^2
}.
\]

Therefore,
\[
C =
\begin{bmatrix}
1 & 0 & 0 & 0 \\[0.4em]
\rho_{12} & \sqrt{1-\rho_{12}^2} & 0 & 0 \\[0.6em]
\rho_{13} &
\dfrac{\rho_{23}-\rho_{12}\rho_{13}}{\sqrt{1-\rho_{12}^2}} &
\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
} & 0 \\[1em]
\rho_{14} &
\dfrac{\rho_{24}-\rho_{12}\rho_{14}}{\sqrt{1-\rho_{12}^2}} &
\dfrac{
\rho_{34}-\rho_{13}\rho_{14}
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})(\rho_{24}-\rho_{12}\rho_{14})}{1-\rho_{12}^2}
}{
\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
} &
c_{44}
\end{bmatrix},
\]
with \(c_{44}\) given above. Hence,
\[
L = D C.
\]

\subsection{Final expressions for the volatility matrix}

Collecting the results, the Cholesky factor \(L\) used in the model is:

\paragraph{Dimension \(2\):}
\[
L =
\begin{bmatrix}
\sigma_1 & 0 \\
\rho_{12}\sigma_2 & \sigma_2\sqrt{1-\rho_{12}^2}
\end{bmatrix}.
\]

\paragraph{Dimension \(3\):}
\[
L =
\begin{bmatrix}
\sigma_1 & 0 & 0 \\
\rho_{12}\sigma_2 & \sigma_2\sqrt{1-\rho_{12}^2} & 0 \\
\rho_{13}\sigma_3 &
\sigma_3\dfrac{\rho_{23}-\rho_{12}\rho_{13}}{\sqrt{1-\rho_{12}^2}} &
\sigma_3\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
\end{bmatrix}.
\]

\paragraph{Dimension \(4\):}
\[
L =
\begin{bmatrix}
\sigma_1 & 0 & 0 & 0 \\[0.4em]
\rho_{12}\sigma_2 & \sigma_2\sqrt{1-\rho_{12}^2} & 0 & 0 \\[0.6em]
\rho_{13}\sigma_3 &
\sigma_3\dfrac{\rho_{23}-\rho_{12}\rho_{13}}{\sqrt{1-\rho_{12}^2}} &
\sigma_3\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
} & 0 \\[1em]
\rho_{14}\sigma_4 &
\sigma_4\dfrac{\rho_{24}-\rho_{12}\rho_{14}}{\sqrt{1-\rho_{12}^2}} &
\sigma_4\dfrac{
\rho_{34}-\rho_{13}\rho_{14}
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})(\rho_{24}-\rho_{12}\rho_{14})}{1-\rho_{12}^2}
}{
\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
} &
\sigma_4 c_{44}
\end{bmatrix},
\]
where
\[
c_{44}
=
\sqrt{
1-\rho_{14}^2
-\frac{(\rho_{24}-\rho_{12}\rho_{14})^2}{1-\rho_{12}^2}
-\left(
\frac{
\rho_{34}-\rho_{13}\rho_{14}
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})(\rho_{24}-\rho_{12}\rho_{14})}{1-\rho_{12}^2}
}{
\sqrt{
1-\rho_{13}^2
-\dfrac{(\rho_{23}-\rho_{12}\rho_{13})^2}{1-\rho_{12}^2}
}
}
\right)^2
}.
\]

\subsection{Remark}

The formulas above give the Cholesky factor associated with a valid correlation matrix \(R\). In dimension \(2\), \(|\rho_{12}|<1\) is sufficient for positive definiteness. In dimensions \(3\) and \(4\), the pairwise bounds \(\rho_{ij}\in(-1,1)\) are not sufficient; the full matrix \(R\) must also be positive definite. Thus, for \(\ell>2\), additional validation or a positive-definite parameterization is required.

\section{Hyperspherical Cholesky Parameterisation}
\label{app:hyperspherical_cholesky}

The standard Cholesky factorisation constructs a lower-triangular factor from a matrix that is already positive definite. In the stable model, we instead need a parameterisation that guarantees positive definiteness by construction. This can be achieved by constructing the Cholesky factor directly using hyperspherical coordinates, following the correlation-matrix representation of \citet{PourahmadiWang2015}.

Let \(R\in\mathbb{R}^{\ell\times\ell}\) denote the matrix constructed from the
hyperspherical Cholesky factor. We construct a lower-triangular matrix \(L_R\)
and define
\[
R=L_RL_R^\top.
\]
The rows of \(L_R\) are parameterised as unit vectors. For \(\ell=3\), for
example, we set
\[
L_R
=
\begin{pmatrix}
1 & 0 & 0 \\
\cos\theta_{21} & \sin\theta_{21} & 0 \\
\cos\theta_{31} &
\sin\theta_{31}\cos\theta_{32} &
\sin\theta_{31}\sin\theta_{32}
\end{pmatrix},
\]
where
\[
\theta_{21},\theta_{31},\theta_{32}\in(0,\pi).
\]
Each row has Euclidean norm one. Moreover, since all diagonal entries of
\(L_R\) are strictly positive, \(L_R\) has full rank.

More generally, for \(i=1,\ldots,\ell\), the \(i\)-th row of \(L_R\) is
constructed using \(i-1\) angles. The first row is
\[
(L_R)_{11}=1.
\]
For \(i\geq2\), the nonzero entries are
\[
(L_R)_{i1}=\cos\theta_{i1},
\]
\[
(L_R)_{ij}
=
\left(\prod_{k=1}^{j-1}\sin\theta_{ik}\right)
\cos\theta_{ij},
\qquad
j=2,\ldots,i-1,
\]
and
\[
(L_R)_{ii}
=
\prod_{k=1}^{i-1}\sin\theta_{ik}.
\]
All entries above the diagonal are zero. If all angles satisfy
\[
\theta_{ij}\in(0,\pi),
\qquad
i=2,\ldots,\ell,\quad j=1,\ldots,i-1,
\]
then all sine terms are strictly positive. Hence the diagonal entries of
\(L_R\) are strictly positive, so \(L_R\) is full rank.

In the neural implementation, the network outputs a flat vector. The first
\(\ell\) outputs are used for the volatility levels, while the remaining
\(\ell(\ell-1)/2\) outputs are used for the correlation structure. We assign
these correlation outputs to the triangular angle indices by defining
\[
q(i,j)=\frac{(i-1)(i-2)}{2}+j,
\qquad
i=2,\ldots,\ell,\quad j=1,\ldots,i-1.
\]
Thus the unconstrained network output corresponding to angle \(\theta_{ij}\)
is
\[
g_{ij}(z)
=
[\mathcal{H}(z)]_{\ell+q(i,j)}.
\]
The corresponding angle is generated by
\[
\theta_{ij}(z)
=
\pi\,\operatorname{sigmoid}(g_{ij}(z)).
\]
Since \(\operatorname{sigmoid}(g_{ij}(z))\in(0,1)\), this maps every
unconstrained network output into
\[
\theta_{ij}(z)\in(0,\pi).
\]
Thus all angles used in the hyperspherical construction lie in \((0,\pi)\), so
the diagonal entries of \(L_R(z)\) are strictly positive for every latent state
\(z\).

\subsection{Validity of the constructed correlation matrix}
\label{app:hyperspherical_cholesky_validity}

We now verify that the matrix constructed above is a valid correlation matrix.
Recall that
\[
R=L_RL_R^\top.
\]
First, \(R\) is symmetric, since
\[
R^\top
=
\left(L_RL_R^\top\right)^\top
=
L_RL_R^\top
=
R.
\]

Second, the diagonal entries of \(R\) are equal to one. The \(i\)-th diagonal
entry of \(R\) is the inner product of the \(i\)-th row of \(L_R\) with itself:
\[
R_{ii}
=
(L_R)_{i,\cdot}(L_R)_{i,\cdot}^\top
=
\|(L_R)_{i,\cdot}\|^2.
\]
For \(i=1\), this is immediate because the first row is
\[
(1,0,\ldots,0).
\]
For \(i\geq2\), the hyperspherical construction gives
\[
\|(L_R)_{i,\cdot}\|^2
=
\cos^2\theta_{i1}
+
\sum_{j=2}^{i-1}
\left(
\prod_{k=1}^{j-1}\sin^2\theta_{ik}
\right)
\cos^2\theta_{ij}
+
\prod_{k=1}^{i-1}\sin^2\theta_{ik}.
\]
This expression equals one by repeated use of the identity
\[
\cos^2 x+\sin^2 x=1.
\]
Equivalently, it can be written recursively as
\[
\cos^2\theta_{i1}
+
\sin^2\theta_{i1}
\left[
\cos^2\theta_{i2}
+
\sin^2\theta_{i2}
\left[
\cdots
\left[
\cos^2\theta_{i,i-1}
+
\sin^2\theta_{i,i-1}
\right]
\cdots
\right]
\right]
=
1.
\]
Hence each row of \(L_R\) has unit norm, and therefore
\[
R_{ii}=1,
\qquad
i=1,\ldots,\ell.
\]

Third, \(R\) is positive definite. Since all angles satisfy
\[
\theta_{ij}\in(0,\pi),
\]
we have
\[
\sin\theta_{ij}>0.
\]
The diagonal entries of \(L_R\) are therefore strictly positive, because
\[
(L_R)_{ii}
=
\prod_{k=1}^{i-1}\sin\theta_{ik}
>
0,
\qquad
i=2,\ldots,\ell,
\]
and
\[
(L_R)_{11}=1.
\]
Since \(L_R\) is lower triangular with nonzero diagonal entries, \(L_R\) has
full rank.

Now take any nonzero vector \(x\in\mathbb{R}^{\ell}\). Then
\[
x^\top R x
=
x^\top L_RL_R^\top x
=
\left(L_R^\top x\right)^\top
\left(L_R^\top x\right)
=
\|L_R^\top x\|^2.
\]
Because \(L_R\) has full rank, \(L_R^\top x\neq 0\) whenever \(x\neq 0\).
Therefore
\[
x^\top R x
=
\|L_R^\top x\|^2
>
0.
\]
Thus \(R\) is positive definite.

Combining these properties, \(R\) is symmetric, positive definite, and has unit
diagonal entries. Hence \(R\) is a valid correlation matrix by construction.

\section{Numerical Solution of the ODE System Using the Runge--Kutta 3/8 Method}
\label{app:rk38}

In the affine-inspired encoder--decoder framework, the bond pricing functions are obtained as the solution of a system of ordinary differential equations (ODEs). Closed-form solutions are generally not available once the coefficient functions are represented by neural networks. Consequently, the system must be solved numerically.

To this end, we employ a fourth-order explicit Runge--Kutta scheme, specifically the \emph{Runge--Kutta 3/8 method}. This method offers a good balance between numerical accuracy and computational efficiency, while remaining straightforward to implement within automatic differentiation frameworks used for neural network training.

\subsection{General ODE formulation}

Consider an ODE of the form
\[
\frac{dy}{dt} = f(t,y),
\qquad
y(t_0) = y_0,
\]
where \(y(t)\in\mathbb{R}^n\) and \(f\) is a sufficiently smooth function. The goal is to compute an approximation of the solution at discrete time points \(t_n = t_0 + nh\), where \(h\) denotes the step size.

\subsection{Runge--Kutta methods}

Runge--Kutta methods approximate the solution by evaluating the derivative function \(f(t,y)\) at several intermediate points within each step. The classical fourth-order Runge--Kutta family achieves local truncation error of order \(O(h^5)\) and global error of order \(O(h^4)\).

Among these schemes, the Runge--Kutta 3/8 rule is a particular fourth-order method characterized by a symmetric weighting structure. It is defined by four intermediate slope evaluations.

\subsection{Runge--Kutta 3/8 scheme}

Let \(y_n\) denote the approximation to \(y(t_n)\). The Runge--Kutta 3/8 update from \(t_n\) to \(t_{n+1}=t_n+h\) is given by

\[
k_1 = f(t_n, y_n),
\]

\[
k_2 = f\left(t_n+\frac{h}{3},\; y_n + \frac{h}{3}k_1\right),
\]

\[
k_3 = f\left(t_n+\frac{2h}{3},\; y_n + h\left(-\frac{1}{3}k_1 + k_2\right)\right),
\]

\[
k_4 = f\left(t_n+h,\; y_n + h\left(k_1 - k_2 + k_3\right)\right).
\]

The solution is then updated according to

\[
y_{n+1}
=
y_n
+
\frac{h}{8}
\left(
k_1 + 3k_2 + 3k_3 + k_4
\right).
\]

This scheme achieves fourth-order accuracy while requiring only four evaluations of the function \(f\) per step.

\subsection{Application in the model}

In the implementation of the affine-inspired encoder--decoder model, the Runge--Kutta 3/8 scheme is used to numerically integrate the ODE system governing the bond pricing functions. The maturity variable plays the role of the time variable in the ODE, and the integration proceeds along the maturity dimension.

Because the scheme is explicit and differentiable, it integrates naturally with modern deep learning frameworks such as PyTorch. Automatic differentiation can therefore propagate gradients through the numerical solver during model training.

The Runge--Kutta 3/8 method provides a fourth-order explicit approximation to the ODE solution while remaining compatible with gradient-based optimization.

\section{Well-posedness of the Stable Latent Dynamics}
\label{app:stable_sde_wellposed}

This appendix verifies that the stable latent dynamics introduced in
Chapter~\ref{chap:simulation} define a well-posed stochastic process. In
particular, we show that the drift and diffusion coefficients satisfy the
global Lipschitz and linear growth conditions of
Proposition~\ref{prop:sde_existence}. It follows that the stable latent SDE
admits a unique non-explosive solution.

The stable latent dynamics are given by
\begin{equation}
    dz_t
    =
    \mu_{\mathrm{st}}(z_t)\,dt
    +
    L_{\mathrm{st}}(z_t)\,dW_t^{\mathbb Q},
    \label{eq:app_stable_sde}
\end{equation}
where \(z_t\in\mathbb{R}^{\ell}\), \(W^{\mathbb Q}\) is an
\(\ell\)-dimensional Brownian motion, and
\[
    \mu_{\mathrm{st}}(z)
    =
    Mz+N,
    \qquad
    M=-(V^\top V+\varepsilon I_\ell),
    \qquad
    \varepsilon>0.
\]
The diffusion matrix is constructed from bounded volatility levels and a valid
correlation matrix. Let
\[
    a_i(z),
    \qquad
    c_{ij}(z),
    \quad 1\leq j<i\leq \ell,
\]
denote the raw outputs of the stable volatility network corresponding,
respectively, to volatility levels and correlation angles. The volatility
levels are defined by
\begin{equation}
    \log \sigma_i(z)
    =
    \log \sigma_{\min}
    +
    \bigl(\log\sigma_{\max}-\log\sigma_{\min}\bigr)
    \operatorname{sigmoid}(a_i(z)),
    \label{eq:app_stable_vol}
\end{equation}
where \(0<\sigma_{\min}<\sigma_{\max}<\infty\). The raw correlation outputs
are mapped to angles by
\[
    \theta_{ij}(z)
    =
    \pi\,\operatorname{sigmoid}(c_{ij}(z))
    \in(0,\pi).
\]
These angles define the hyperspherical Cholesky factor \(L_R(z)\), and hence
the correlation matrix
\[
    R(z)=L_R(z)L_R(z)^\top.
\]
The diffusion loading matrix is then
\[
    L_{\mathrm{st}}(z)
    =
    D(z)L_R(z),
    \qquad
    D(z)=\operatorname{diag}(\sigma_1(z),\ldots,\sigma_\ell(z)).
\]
Thus
\[
    L_{\mathrm{st}}(z)L_{\mathrm{st}}(z)^\top
    =
    D(z)R(z)D(z)
\]
is the covariance matrix used in the latent SDE.

\begin{proposition}[Well-posedness of the stable latent SDE]
\label{prop:stable_sde_wellposed}
Suppose all neural network weights are finite and that
\[
    \varepsilon>0,
    \qquad
    0<\sigma_{\min}<\sigma_{\max}<\infty.
\]
Then the coefficients \(\mu_{\mathrm{st}}\) and \(L_{\mathrm{st}}\) are
globally Lipschitz and satisfy a linear growth condition. Consequently, for
every initial state \(z_0\in\mathbb{R}^{\ell}\), the SDE
\eqref{eq:app_stable_sde} admits a unique strong solution on every finite time
interval \([0,T]\). The solution is non-explosive and has continuous sample
paths.
\end{proposition}

\begin{proof}
We verify the drift and diffusion conditions separately.

First, consider the drift
\[
    \mu_{\mathrm{st}}(z)=Mz+N.
\]
Since \(V\) and \(N\) are finite learned parameters and \(\varepsilon>0\), the
matrix
\[
    M=-(V^\top V+\varepsilon I_\ell)
\]
is finite, and hence \(\|M\|<\infty\). For any
\(z,y\in\mathbb{R}^{\ell}\),
\[
    \|\mu_{\mathrm{st}}(z)-\mu_{\mathrm{st}}(y)\|
    =
    \|M(z-y)\|
    \leq
    \|M\|\,\|z-y\|.
\]
Thus the drift is globally Lipschitz. Moreover,
\[
    \|\mu_{\mathrm{st}}(z)\|
    \leq
    \|M\|\,\|z\|+\|N\|
    \leq
    C(1+\|z\|)
\]
for some constant \(C>0\). Hence the drift satisfies a linear growth condition.

We next consider the diffusion. The raw functions \(a_i(z)\) and \(c_{ij}(z)\)
are outputs of neural networks consisting of affine maps and smooth activation
functions with bounded derivatives. For any fixed finite set of network
weights, these raw output maps are therefore globally Lipschitz.

The volatility map in \eqref{eq:app_stable_vol} is a composition of globally
Lipschitz functions. Since the sigmoid takes values in \((0,1)\),
\[
    \log\sigma_i(z)
    \in
    (\log\sigma_{\min},\log\sigma_{\max}),
\]
and therefore
\[
    \sigma_i(z)
    \in
    (\sigma_{\min},\sigma_{\max})
    \qquad
    \text{for all }z\in\mathbb{R}^{\ell}.
\]
Because the exponential function has bounded derivative on the compact
interval
\[
    [\log\sigma_{\min},\log\sigma_{\max}],
\]
each map \(z\mapsto\sigma_i(z)\) is globally Lipschitz and bounded.

The hyperspherical angles satisfy
\[
    \theta_{ij}(z)
    =
    \pi\,\operatorname{sigmoid}(c_{ij}(z)).
\]
They are therefore globally Lipschitz as compositions of globally Lipschitz
functions. The entries of \(L_R(z)\) are finite products of sine and cosine
functions evaluated at these angles. Since sine and cosine have bounded
derivatives, and products of bounded Lipschitz functions are Lipschitz, the map
\(z\mapsto L_R(z)\) is globally Lipschitz. Furthermore, the entries of
\(L_R(z)\) are bounded by construction.

It follows that
\[
    L_{\mathrm{st}}(z)=D(z)L_R(z)
\]
is a product of bounded globally Lipschitz matrix-valued functions. Hence
\(L_{\mathrm{st}}\) is globally Lipschitz. Moreover, since the volatility
levels are bounded and \(L_R(z)\) has bounded entries, there exists a constant
\(C_L>0\) such that
\[
    \|L_{\mathrm{st}}(z)\|
    \leq
    C_L
    \leq
    C_L(1+\|z\|)
    \qquad
    \text{for all }z\in\mathbb{R}^{\ell}.
\]
Thus the diffusion also satisfies a linear growth condition.

The drift and diffusion coefficients therefore satisfy the assumptions of
Proposition~\ref{prop:sde_existence}. Consequently, the stable latent SDE
\eqref{eq:app_stable_sde} admits a unique strong solution on every finite
horizon. The linear growth condition rules out finite-time explosion, so the
stable latent dynamics are well defined.
\end{proof}

The stable drift also has an explicit mean-reverting structure. For any
non-zero \(x\in\mathbb{R}^{\ell}\),
\[
    x^\top(V^\top V+\varepsilon I_\ell)x
    =
    \|Vx\|^2+\varepsilon\|x\|^2
    >
    0.
\]
Therefore \(V^\top V+\varepsilon I_\ell\) is symmetric positive definite, and
all eigenvalues of
\[
    M=-(V^\top V+\varepsilon I_\ell)
\]
are strictly negative. The deterministic part of the stable dynamics is
therefore globally mean-reverting around the fixed point
\[
    z^*=-M^{-1}N.
\]

Finally, the short-rate function in the stable model is bounded by
construction. It is specified as
\[
    r(z)
    =
    r_{\mathrm{center}}
    +
    r_{\mathrm{scale}}\tanh(f(z)),
\]
where
\[
    r_{\mathrm{scale}}
    =
    \operatorname{softplus}(\eta)+s_{\min}
    >
    0.
\]
Since \(\tanh(f(z))\in(-1,1)\), we have
\[
    r(z)
    \in
    \bigl(
    r_{\mathrm{center}}-r_{\mathrm{scale}},
    \;
    r_{\mathrm{center}}+r_{\mathrm{scale}}
    \bigr).
\]
Consequently, for every finite horizon \(T\), the simulated discount factor
\[
    D_T
    =
    \exp\left(-\int_0^T r(z_s)\,ds\right)
\]
is finite and strictly positive. Thus both the latent paths and the associated
discount factors used in the simulation experiment are mathematically
well-defined.
