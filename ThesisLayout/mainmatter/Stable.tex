\chapter{Towards a Stable Autoencoder}
\label{chap:stable_ae}


The autoencoder implemented optimizes for reconstruction fit, and not for stability. 

Even though the drift and diffusion look reasonable, the balance between the drift and diffusion matters. 


We first need to check if this overholds the condition for existence and uniqueness of an SDE.

\begin{theorem}
\label{thm:existence_uniqueness_sde}
 Let $\mu:\mathbb{R}^\ell\rightarrow\mathbb{R}^\ell$ and $\sigma:\mathbb{R}^\ell\rightarrow\mathbb{R}^{\ell\times\ell}$, be functions where
$$|\mu(z)|+|\sigma(z)|\leq C(1+|z|),\;\;\forall z\in\mathbb{R}^\ell$$

for some constant $C$. Furthermore if the Lipschitz condition holds

$$|\mu(z_1)-\mu(z_2)|+|\sigma(z_1)-\sigma(z_2)|\leq D|z_1-z_2|,\;\;\forall z_1, z_2,\in\mathbb{R}^\ell$$

for some constant $D$. Then the SDE
$$dz_t=\mu(z_t)dt+\sigma(z_t)dW_t$$
Has a unique solution, $z_t$, which is adapted to the filtration generated by $W$ and
$$E\left[\int_0^T||z_t||^2dt\right]<\infty$$       
\end{theorem}




So under this two conditions, the process is well-behaved, unique, and cannot explode. So we will now check if our Neural Networks actually obide this


\begin{proposition}
Let $\mathcal{K}: \mathbb{R}^\ell \to \mathbb{R}^\ell$ and 
$\mathcal{H}: \mathbb{R}^\ell \to \mathbb{R}^{\ell + \ell(\ell-1)/2}$ denote the drift 
and volatility networks of the baseline model. Then both satisfy the linear growth 
and Lipschitz conditions of Theorem~\ref{thm:existence_uniqueness_sde}.
\end{proposition}

\begin{proof}
\textbf{Drift $\mathcal{K}$.}
The drift network performs the affine transformation
\[
\mathcal{K}(z) = Mz + N, \quad M \in \mathbb{R}^{\ell \times \ell},\; N \in \mathbb{R}^\ell.
\]
\emph{Linear growth.} By the triangle inequality and submultiplicativity of the operator norm:
\[
\|\mathcal{K}(z)\| = \|Mz + N\| \leq \|M\|\|z\| + \|N\| \leq \max(\|M\|, \|N\|)(1 + \|z\|).
\]
So linear growth holds with $C_1 = \max(\|M\|, \|N\|)$.

\emph{Lipschitz.} For any $z, z' \in \mathbb{R}^\ell$:
\[
\|\mathcal{K}(z) - \mathcal{K}(z')\| = \|M(z - z')\| \leq \|M\|\|z - z'\|.
\]
So Lipschitz holds with $L_1 = \|M\|$.

\textbf{Diffusion $\mathcal{H}$.}
The volatility network has one hidden layer with no bias,
\[
\mathcal{H}(z) = W_2\,\psi(W_1 z), \quad 
W_1 \in \mathbb{R}^{h \times \ell},\; 
W_2 \in \mathbb{R}^{(\ell + \ell(\ell-1)/2) \times h},
\]
where $\psi(x) = \mathrm{sigmoid}(x) - 0.5$ is the CenteredSoftStep activation 
and $h$ denotes the hidden dimension. The actual volatilities are recovered as 
$\sigma_i(z) = \exp([\mathcal{H}(z)]_i)$.

\emph{Step 1: Bound the activation.} Since $\mathrm{sigmoid}(x) \in (0,1)$ for all 
$x \in \mathbb{R}$:
\[
|\psi(x)| = |\mathrm{sigmoid}(x) - 0.5| < 0.5 \quad \forall\, x \in \mathbb{R},
\]
and therefore $\|\psi(W_1 z)\| \leq \frac{\sqrt{h}}{2}$ for all $z \in \mathbb{R}^\ell$.

\emph{Step 2: Bound the network output.} By submultiplicativity:
\[
\|\mathcal{H}(z)\| = \|W_2\,\psi(W_1 z)\| \leq \|W_2\|\cdot\frac{\sqrt{h}}{2} 
=: C_{\mathrm{pre}} < \infty.
\]
This bound is independent of $z$, so each component satisfies 
$[\mathcal{H}(z)]_i \in [-C_{\mathrm{pre}},\, C_{\mathrm{pre}}]$ for all $z$.

\emph{Step 3: Linear growth of $\sigma_\theta$.} Since $\exp$ is monotone increasing:
\[
\|\sigma_\theta(z)\| = \bigl\|\exp(\mathcal{H}(z))\bigr\| 
\leq \sqrt{\ell}\,e^{C_{\mathrm{pre}}} 
\leq \sqrt{\ell}\,e^{C_{\mathrm{pre}}}(1 + \|z\|).
\]
So linear growth holds with $C_2 = \sqrt{\ell}\,e^{C_{\mathrm{pre}}}$.

\emph{Step 4: Lipschitz continuity of $\sigma_\theta$.} For any $z, z' \in \mathbb{R}^\ell$, 
apply the mean value theorem to $\exp$. Since 
$\exp'(x) = \exp(x) \leq e^{C_{\mathrm{pre}}}$ on $[-C_{\mathrm{pre}}, C_{\mathrm{pre}}]$:
\[
|\sigma_i(z) - \sigma_i(z')| \leq e^{C_{\mathrm{pre}}}|[\mathcal{H}(z)]_i - [\mathcal{H}(z')]_i|.
\]
Summing over components and applying submultiplicativity, together with the fact that 
$\psi'(x) = \mathrm{sigmoid}(x)(1 - \mathrm{sigmoid}(x)) \leq \frac{1}{4}$:
\[
\|\sigma_\theta(z) - \sigma_\theta(z')\| 
\leq e^{C_{\mathrm{pre}}}\|\mathcal{H}(z) - \mathcal{H}(z')\| 
\leq e^{C_{\mathrm{pre}}}\|W_2\|\cdot\frac{\|W_1\|}{4}\cdot\|z - z'\|.
\]
So Lipschitz holds with $L_2 = \frac{e^{C_{\mathrm{pre}}}\|W_1\|\|W_2\|}{4}$.

\textbf{Conclusion.} Both $\mathcal{K}$ and $\mathcal{H}$ satisfy the linear growth and 
Lipschitz conditions with explicit constants
\[
C_1 = \max(\|M\|, \|N\|), \quad L_1 = \|M\|, \quad
C_2 = \sqrt{\ell}\,e^{C_{\mathrm{pre}}}, \quad 
L_2 = \frac{e^{C_{\mathrm{pre}}}\|W_1\|\|W_2\|}{4}, \quad 
C_{\mathrm{pre}} = \frac{\sqrt{h}}{2}\|W_2\|.
\]
Therefore, by Theorem~\ref{thm:existence_uniqueness_sde}, the baseline model SDE has a 
unique, non-explosive solution. The key architectural property enabling this result 
is the bounded activation $\psi \in (-0.5, 0.5)$, which confines the pre-exponential 
outputs of $\mathcal{H}$ to a bounded interval independent of $z$, preventing the 
diffusion from growing with the state.
\end{proof}


\section{Problem 2: Mean reversion?}

Then the next problem is if the drift is mean reverting, because nothing in the baseline model is guaranteeing that. 



\begin{proposition}
The drift $\mathcal{K}(z) = Mz + N$ is mean-reverting toward the fixed point
$z^* = -M^{-1}N$ if and only if all eigenvalues of $M$ have strictly negative
real parts.
\end{proposition}

\begin{proof}
\textbf{Setup.} Assume $M$ is invertible so that $z^* = -M^{-1}N$ is 
well-defined. Define the deviation process:
\[
u_t = z_t - z^*
\]
Substituting into the ODE $\dot{z}_t = Mz_t + N$:
\[
\dot{u}_t = Mz_t + N = M(u_t + z^*) + N = Mu_t + 
\underbrace{Mz^* + N}_{=0} = Mu_t
\]
So the dynamics reduce to the homogeneous linear ODE:
\[
\dot{u}_t = Mu_t, \quad u_0 = z_0 - z^*
\]
By \citet[][p.~129]{Hirsch2012}, the unique solution is:
\[
u_t = e^{Mt}u_0
\]
where $e^{Mt}$ denotes the matrix exponential.

\textbf{($\Rightarrow$)} Suppose all eigenvalues satisfy 
$\mathrm{Re}(\lambda_i) < 0$. By the Jordan decomposition, write 
$M = PJP^{-1}$ where $J$ is the Jordan normal form, so that 
$e^{Mt} = Pe^{Jt}P^{-1}$. Each Jordan block $J_k$ corresponding to 
eigenvalue $\lambda_k$ contributes terms of the form $t^j e^{\lambda_k t}$ 
for $j = 0, 1, \ldots$ Since $\mathrm{Re}(\lambda_k) < 0$ for all $k$, 
polynomial growth is dominated by exponential decay:
\[
|t^j e^{\lambda_k t}| = t^j e^{\mathrm{Re}(\lambda_k)t} \to 0 
\quad \text{as } t \to \infty.
\]
Therefore $e^{Jt} \to 0$ and hence $e^{Mt} \to 0$, which gives:
\[
\|u_t\| = \|e^{Mt}u_0\| \leq \|P\|\|e^{Jt}\|\|P^{-1}\|\|u_0\| \to 0.
\]
So $z_t = z^* + u_t \to z^*$, i.e.\ $z^*$ is asymptotically stable. This 
is consistent with \citet[][p.~175]{Hirsch2012}, who establish that an 
equilibrium point is asymptotically stable if and only if all eigenvalues 
of the system matrix have strictly negative real parts. Since 
$\mathcal{K}$ is already linear, no linearization is needed and the 
result applies directly.

\textbf{($\Leftarrow$)} Suppose some eigenvalue satisfies 
$\mathrm{Re}(\lambda_i) \geq 0$, and let $v_i$ be the corresponding 
eigenvector. Choose $u_0 = v_i$. Then:
\[
u_t = e^{Mt}v_i = e^{\lambda_i t}v_i
\]
and therefore:
\[
\|u_t\| = |e^{\lambda_i t}|\|v_i\| = 
e^{\mathrm{Re}(\lambda_i)t}\|v_i\| \geq \|v_i\| > 0 
\quad \forall\, t \geq 0.
\]
So $u_t \not\to 0$, meaning $z_t \not\to z^*$.

\textbf{Conclusion.} The drift $\mathcal{K}(z) = Mz + N$ is mean-reverting 
toward $z^*$ if and only if all eigenvalues of $M$ have strictly negative 
real parts:
\[
\mathrm{Re}(\lambda_i) < 0 \quad \forall\, i = 1, \ldots, \ell.
\]
This is equivalent to $M$ being a \emph{Hurwitz matrix} 
\citep[][p.~175]{Hirsch2012}. Since the baseline model places no 
constraint on $M$ during training, mean reversion is not guaranteed --- 
whether it holds depends entirely on the learned parameters.
\end{proof}

\subsection*{Enforcing the Hurwitz Condition}

While the Hurwitz condition $\mathrm{Re}(\lambda_i) < 0$ for all $i$ is 
necessary and sufficient for mean reversion of the drift, it is not 
straightforward to enforce directly during training, as it would require 
monitoring the eigenvalues of $M$ at every gradient step. Instead, we 
construct $M$ such that it is Hurwitz \emph{by construction}.

\begin{proposition}
Let $V \in \mathbb{R}^{\ell \times \ell}$ be any matrix and $\epsilon > 0$. 
Then the matrix
\[
M = -(V^\top V + \epsilon I)
\]
is Hurwitz.
\end{proposition}

\begin{proof}
We show that all eigenvalues of $M$ have strictly negative real parts.
Let $\lambda$ be any eigenvalue of $M$ with corresponding eigenvector 
$x \neq 0$. Then:
\[
Mx = \lambda x \implies -(V^\top V + \epsilon I)x = \lambda x
\]
Taking the inner product with $x$:
\[
-x^\top(V^\top V + \epsilon I)x = \lambda \|x\|^2
\]
Since $V^\top V$ is positive semi-definite (as $x^\top V^\top V x = 
\|Vx\|^2 \geq 0$ for all $x$) and $\epsilon I$ is strictly positive 
definite, we have:
\[
x^\top(V^\top V + \epsilon I)x = \|Vx\|^2 + \epsilon\|x\|^2 \geq 
\epsilon\|x\|^2 > 0
\]
Therefore:
\[
\lambda = -\frac{x^\top(V^\top V + \epsilon I)x}{\|x\|^2} \leq 
-\epsilon < 0
\]
Since this holds for any eigenvector $x$, all eigenvalues of $M$ satisfy 
$\lambda \leq -\epsilon < 0$, and in particular 
$\mathrm{Re}(\lambda) \leq -\epsilon < 0$. Hence $M$ is Hurwitz. 
\end{proof}

\begin{remark}
Note that $M = -(V^\top V + \epsilon I)$ is always symmetric and negative 
definite. This is a sufficient but not necessary condition for being 
Hurwitz --- a general Hurwitz matrix need not be symmetric. The 
parameterization is therefore slightly more restrictive than the general 
Hurwitz condition, but has the practical advantage that $M$ is guaranteed 
to be Hurwitz for any choice of $V$, allowing $V$ to be freely learned 
during training without any additional constraints. The parameter 
$\epsilon > 0$ ensures strict negative definiteness even when $V$ is 
singular, and controls the minimum rate of mean reversion.
\end{remark}

\section{Problem 3: Bounded and positive-definite diffusion}
\label{sec:stable_diffusion}
%========================================================

Even when the drift is mean-reverting, the Euler--Maruyama simulation can still produce implausible paths if the diffusion coefficient grows too large or if the correlation matrix ceases to be positive definite at states far from the training cloud.
The baseline diffusion network $\mathcal{H}$ exhibits two potential weaknesses that become practically relevant during long-horizon simulation.

\subsection{Unbounded volatilities}

In the baseline parameterization, the individual volatilities are recovered as
\[
\sigma_i(z) = \exp\bigl([\mathcal{H}(z)]_i\bigr), \qquad i = 1,\ldots,\ell.
\]
We showed in the proof of Proposition~\ref{thm:existence_uniqueness_sde} that the bounded activation $\psi$ confines the pre-exponential outputs to a compact interval $[-C_{\mathrm{pre}}, C_{\mathrm{pre}}]$, so that $\sigma_i \in [e^{-C_{\mathrm{pre}}}, e^{C_{\mathrm{pre}}}]$ for all $z$.
This guarantees the theoretical linear-growth and Lipschitz conditions needed for existence and uniqueness of the SDE solution, but the \emph{practical} upper bound $e^{C_{\mathrm{pre}}}$ depends on the learned weight matrices $W_1,W_2$ and can be large.
When the latent state drifts outside the support of the training data --- as inevitably happens during multi-year simulation --- the network may output pre-exponential values near its saturation limits, producing volatilities that are technically finite but economically implausible (e.g.\ $\sigma_i > 5$).

To address this, we replace the $\exp$-link with a \emph{sigmoid-bounded} parameterization:
\[
\log\sigma_i(z) \;=\; \log\sigma_{\min}
  + \bigl(\log\sigma_{\max} - \log\sigma_{\min}\bigr)
    \cdot \operatorname{sigmoid}\!\bigl(h_i(z) + b_i\bigr),
\]
where $h_i(z)$ denotes the $i$-th output of the shared hidden layer, $b_i$ is a learnable offset initialised so that $\sigma_i = \sigma_{\mathrm{init}}$ when $h_i = 0$, and $0 < \sigma_{\min} < \sigma_{\max}$ are user-specified bounds.
This guarantees
\[
\sigma_i(z) \;\in\; (\sigma_{\min},\;\sigma_{\max}) \qquad \forall\, z \in \mathbb{R}^\ell,
\]
while retaining full differentiability. The sigmoid link is monotone and its derivative is bounded, so the Lipschitz property established in the proof above still holds --- in fact with a \emph{tighter} constant, since the output range is now strictly controlled.

\begin{proposition}
\label{prop:bounded_sigma}
Let $\sigma_{\min}, \sigma_{\max} > 0$ with $\sigma_{\min} < \sigma_{\max}$. The stable volatility parameterization satisfies, for all $z \in \mathbb{R}^\ell$:
\begin{enumerate}
    \item $\sigma_i(z) \in (\sigma_{\min}, \sigma_{\max})$ for every $i = 1,\ldots,\ell$.
    \item The mapping $z \mapsto \sigma_i(z)$ is Lipschitz continuous with constant
    \[
    L_{\sigma} = \frac{1}{4}\,(\log\sigma_{\max} - \log\sigma_{\min})\,\sigma_{\max}\,\|W_2^{(i)}\|\,\|W_1\|,
    \]
    where $W_2^{(i)}$ denotes the $i$-th row of the output weight matrix.
\end{enumerate}
\end{proposition}

\begin{proof}
\emph{Part 1.} Since $\operatorname{sigmoid}(x) \in (0,1)$ for all $x \in \mathbb{R}$, we have $\log\sigma_i \in (\log\sigma_{\min},\,\log\sigma_{\max})$ and therefore $\sigma_i \in (\sigma_{\min},\,\sigma_{\max})$.

\emph{Part 2.} Write $s_i(z) = h_i(z) + b_i$ and $\phi = \operatorname{sigmoid}$. Then $\sigma_i = \exp\!\bigl(\log\sigma_{\min} + \Delta\,\phi(s_i)\bigr)$ with $\Delta = \log\sigma_{\max} - \log\sigma_{\min}$. By the chain rule:
\[
\|\nabla_z \sigma_i\| = \sigma_i \cdot \Delta \cdot \phi'(s_i) \cdot \|\nabla_z h_i\|
\leq \sigma_{\max} \cdot \Delta \cdot \tfrac{1}{4} \cdot \|W_2^{(i)}\|\,\|W_1\|,
\]
where we used $\sigma_i \leq \sigma_{\max}$, $\phi'(x) = \phi(x)(1-\phi(x)) \leq \frac{1}{4}$, and the fact that $\|\nabla_z h_i\| \leq \|W_2^{(i)}\|\,\|W_1\|$ because the hidden activation $\psi$ satisfies $|\psi'| \leq \frac{1}{4} \leq 1$.
\end{proof}


\subsection{Positive-definite correlations for arbitrary dimension}

The baseline network parameterizes correlations as
\[
\rho_{ij}(z) = \tanh\!\bigl([\mathcal{H}(z)]_k\bigr),
\]
which ensures each pairwise correlation lies in $(-1,1)$, but does \emph{not} guarantee that the full $\ell \times \ell$ correlation matrix $R$ is positive definite when $\ell \geq 3$.
A set of pairwise correlations $\{\rho_{ij}\}$ that are individually valid can still fail to be jointly realisable: for example, $\rho_{12} = \rho_{13} = 0.9$ and $\rho_{23} = -0.9$ gives a correlation matrix with a negative eigenvalue.
If $R$ is not positive definite, the Cholesky factorization $R = L_R L_R^\top$ fails and the diffusion matrix $\Sigma = \operatorname{diag}(\sigma)\,L_R$ is undefined.

For $\ell = 2$, this issue does not arise: any single correlation $\rho_{12} \in (-1,1)$ gives a valid $2 \times 2$ correlation matrix.
However, since the architecture is designed to work for any $\ell$, a structurally safe parameterization is required.

We adopt the \emph{hyperspherical Cholesky} parameterization \citep{Pinheiro1996}, which maps $\ell(\ell-1)/2$ free angles $\theta_k \in (0,\pi)$ to a lower-triangular matrix $L_R$ such that $R = L_R L_R^\top$ is a valid correlation matrix by construction.

\begin{definition}[Hyperspherical Cholesky factor]
\label{def:hyperspherical_cholesky}
For $\ell \geq 2$, define the lower-triangular matrix $L_R \in \mathbb{R}^{\ell \times \ell}$ by:
\[
L_R[0,0] = 1, \qquad
L_R[i,j] = \cos\theta_{i,j+1}\prod_{k=1}^{j}\sin\theta_{i,k}, \quad j < i, \qquad
L_R[i,i] = \prod_{k=1}^{i}\sin\theta_{i,k},
\]
where the angles $\{\theta_{i,k}\}$ are indexed row by row, using $\ell(\ell-1)/2$ free parameters in total.
\end{definition}

\begin{proposition}
\label{prop:pd_correlation}
If all angles satisfy $\theta_{i,k} \in (0,\pi)$, then $R = L_R L_R^\top$ is a symmetric positive-definite matrix with unit diagonal (i.e.\ a valid correlation matrix).
\end{proposition}

\begin{proof}
\emph{Unit diagonal.} Each row of $L_R$ has squared norm:
\[
\sum_{j=0}^{i} L_R[i,j]^2 = 1,
\]
which follows from the trigonometric identity $\cos^2\theta + \sin^2\theta = 1$ applied recursively.
Therefore $R[i,i] = \sum_j L_R[i,j]^2 = 1$.

\emph{Positive definiteness.} For $\theta_{i,k} \in (0,\pi)$, we have $\sin\theta_{i,k} > 0$, so the diagonal entries $L_R[i,i] = \prod_{k=1}^{i}\sin\theta_{i,k} > 0$.
Hence $L_R$ is lower-triangular with strictly positive diagonal, which means it is non-singular.
For any $x \neq 0$:
\[
x^\top R\, x = x^\top L_R L_R^\top x = \|L_R^\top x\|^2 > 0,
\]
since $L_R^\top$ is non-singular. Therefore $R$ is positive definite.

\emph{Symmetry.} $R = L_R L_R^\top$ is symmetric by construction.
\end{proof}

In the stable network, the angles are produced from the shared hidden layer via
\[
\theta_k(z) = \pi \cdot \operatorname{sigmoid}\!\bigl(g_k(z)\bigr) \;\in\; (0,\pi),
\]
where $g_k(z)$ is the $(d+k)$-th output of the network.
The sigmoid ensures that the angles are always in the open interval $(0,\pi)$, so Proposition~\ref{prop:pd_correlation} applies for every input $z$ and the correlation matrix is guaranteed to be positive definite.

\begin{remark}[Initialization]
When the output-layer weights are initialized to zero, we have $g_k(z) = 0$ for all $z$ and $k$, so $\theta_k = \pi/2$.
Since $\cos(\pi/2) = 0$, the off-diagonal entries of $L_R$ vanish and $L_R = I$, giving $R = I$ (zero correlations).
The model therefore starts from an uncorrelated prior and learns non-trivial correlation structure during training.
\end{remark}

\section{Summary of the stable autoencoder}
\label{sec:stable_summary}
%========================================================

The following table summarises the differences between the baseline and stable variants.
The encoder, decoder $G$, and short-rate network $R$ are \emph{shared} across both variants; only the drift~$K$ and diffusion~$H$ networks differ.

\begin{table}[H]
\centering
\caption{Comparison of baseline and stable network parameterizations.}
\label{tab:stable_summary}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Component} & \textbf{Baseline} & \textbf{Stable} \\
\midrule
Drift matrix $M$ & Unconstrained $M \in \mathbb{R}^{\ell \times \ell}$ & $M = -(V^\top V + \epsilon I)$, Hurwitz by construction \\
Volatilities $\sigma_i$ & $\exp(\text{net}_i)$, bounded but potentially large & $\sigma_i \in (\sigma_{\min}, \sigma_{\max})$ via sigmoid link \\
Correlations $\rho_{ij}$ & $\tanh(\text{net}_k) \in (-1,1)$ per pair & Hyperspherical angles $\to$ PD $R$ by construction \\
\midrule
Mean reversion & Not guaranteed & Guaranteed (all $\lambda_i \leq -\epsilon$) \\
PD diffusion & Guaranteed only for $\ell \leq 2$ & Guaranteed for any $\ell$ \\
$\sigma$ bounds & $\sigma_i \in (0, e^{C_{\text{pre}}})$ & $\sigma_i \in (\sigma_{\min}, \sigma_{\max})$ \\
\bottomrule
\end{tabular}
\end{table}

These modifications ensure that the latent SDE
\[
dz_t = \mathcal{K}(z_t)\,dt + \Sigma(z_t)\,dW_t
\]
is not only well-posed in the sense of Theorem~\ref{thm:existence_uniqueness_sde} (which both variants satisfy), but also \emph{practically stable} during long-horizon Euler--Maruyama simulation: the drift always pulls the state back toward its long-run mean, the volatilities remain in an economically reasonable range, and the diffusion matrix is always well-defined.

The cost of these guarantees is a slight reduction in expressiveness: the stable drift is restricted to symmetric negative-definite matrices rather than arbitrary Hurwitz matrices, and the volatilities cannot exceed the prescribed upper bound.
In the next section, we verify empirically that this trade-off is favourable: the stable variant achieves comparable reconstruction accuracy while producing qualitatively superior simulation behaviour.
