\chapter{Financial Theory}
\label{chap:finance_theory}

This chapter develops the financial theory needed to model the term structure of interest rates. We establish the stochastic calculus and no-arbitrage pricing framework following \citet{Bjork2020}, and the interest rate theory following \citet{Filipovic2009}, leading to the pricing of zero-coupon bonds and the construction of the yield curve.


\section{Stochastic Processes and Calculus}

Consider a continuous time setting with time $t\in[0,T]$, where the processes of interest are modeled as diffusion processes, denoted $X$ throughout. Informally, a stochastic process is a diffusion if its evolution over a small time interval $[t,t+\Delta t]$ can be approximated by 
$$X_{t+\Delta t}-X_t=\mu(t,X_t)\Delta t +\sigma (t,X_t)Z_t$$
where $\mu(t,X_t)$ a locally deterministic drift term, $Z_t$ is a Gaussian disturbance independent of everything that has happened up to time $t$, and $\sigma(t,X_t)$ is the diffusion term scaling the magnitude of the disturbance. The Gaussian disturbance $Z_t$ is modeled through the Brownian motion, defined below.
\begin{definition}[Brownian Motion]
A stochastic process $W$ is a Brownian Motion if the following conditions hold
\begin{enumerate}
    \item $W_0=0$
    \item The process $W$ has independent increments, i.e. if $r<s\leq t <u$ then $W_u-W_t$ and $W_s-W_r$ are independent random variables.
    \item For $s<t$ the random variable $W_t-W_s$ is Gaussian distributed with  mean $0$ and variance $t-s$, i.e. $W_t-W_s\sim\mathcal{N}(0,t-s)$.
    \item $W$ has continuous trajectories.
\end{enumerate}
\label{def:BM}
\end{definition}


Since the stochastic process evolves over time, we must keep track of the information available at each point in time. The information generated by observing $W$ up to time $t$ is captured by the natural filtration $\mathcal{F}_t^W=\sigma(W_s;0\leq s\leq t)$. A stochastic process $X$ is said to be adapted to the filtration $\{\mathcal{F}_t^W\}_{t\geq 0}$ if $X_t$ is $\mathcal{F}_t^W$-measurable for all $t\geq 0$, meaning the value of $X_t$ can be completely determined from the trajectory of $W$ on the interval $[0,t]$.

The diffusion processes mentioned above are formally defined through stochastic differential equations (SDEs), which describe the infinitesimal evolution of $X$ over time. Taking the continuous time limit, the discrete approximation above gives rise to an SDE which in integral form reads
\begin{equation}
    X_t=x_0+\int_0^t\mu_s(X_s)ds+\int_0^t\sigma_s(X_s)dW_s
    \label{eq:SDEIntegral}
\end{equation}
 or equivalently in differential form
\begin{equation}
    \left\{\begin{aligned}
        dX_t &= \mu_t(X_t)dt+\sigma_t(X_t)dW_t \\
        X_0 &= x_0
    \end{aligned}\right.
    \label{eq:SDE}
\end{equation}
where $x_0$ is the initial value of the process. The SDE includes a stochastic integral with respect to $W$, which we establish next.



\subsection{The Stochastic Integral}
We introduce the stochastic integral, also known as the \textit{Itô integral},
\begin{equation}
    \int_a^bg_sdW_s
    \label{ItoIntegral}
\end{equation}
which is well-defined for stochastic process $g$ belonging to the class $\mathcal{L}^2$, defined as follows.

\begin{definition}
    Let $\mathcal{L}^2=\mathcal{L}^2([a,b])$ denote the class of stochastic processes $g$ satisfying:
    \begin{enumerate}
        \item $\int_a^b E[g_s^2]\, ds < \infty$.
        \item $g$ is adapted to the $\mathcal{F}_t^W$-filtration.
    \end{enumerate}
    \label{def:L2}
\end{definition}

The stochastic integral $\int_a^bg_sdW_s$ is well-defined for all $g\in\mathcal{L}^2$, and satisfies the following properties.

\begin{proposition}
    Let $g\in\mathcal{L}^2$. Then
    \begin{enumerate}[label=(\alph*)]
        \item Zero mean:
        \begin{equation}
            E\left[\int_a^b g_s\, dW_s \right] = 0
        \end{equation}
        \item Itô isometry:
        \begin{equation}
            E\left[\left(\int_a^b g_s\, dW_s\right)^2\right] = \int_a^b E[g_s^2]\, ds
        \end{equation}
        \item Measurability: The stochastic integral $\int_a^b g_s\, dW_s$ is $\mathcal{F}_b^W$-measurable.
        \item Conditional zero mean: For any $g\in\mathcal{L}^2[s,t]$
        \begin{equation}
            E\left[\int_s^t g_u\, dW_u \,\middle|\, \mathcal{F}_s^W\right] = 0.
        \end{equation}
    \end{enumerate}
    \label{prop:SIproperties}
\end{proposition}







\subsection{Martingales}

An immediate consequence of Proposition \ref{prop:SIproperties} is that stochastic integrals are martingales.

\begin{definition}[Martingale]
    A stochastic process $X$ is called an $(\mathcal{F}_t)$-martingale if the following conditions hold:
    \begin{enumerate}
        \item $X$ is adapted to the filtration $\{\mathcal{F}_t\}_{t\geq0}$.
        \item For all $t$ we have 
        $$E[|X_t|]<\infty$$
        \item For all $s$ and $t$ with $s\leq t$ the following relation holds
        $$E[X_t|\mathcal{F}_s]=X_s.$$
    \end{enumerate}
    \label{def:martingale}
\end{definition}

A martingale is characterized by the absence of drift. Given the information available at time $s$, the expected value of $X_t$ is simply its current value $X_s$. The following corollary establishes that stochastic integrals are martingales. 

\begin{corollary}
    For any $g\in\mathcal{L}^2$, the process $X_t=\int_0^tg_sdW_s$ is a $\{\mathcal{F}_t^W\}$-martingale.
    \label{cor:SImartingale}
\end{corollary}

\begin{proof}
Let $g\in\mathcal{L}^2$. Since $X_t=\int_0^tg_sdW_s$ is well-defined, we split the integral at $s\leq t$ as
\begin{align*}
    X_t=\int_0^sg_udW_u+\int_s^tg_udW_u=X_s+\int_s^tg_udW_u
\end{align*}
Taking conditional expectations with respect to $\mathcal{F}_s^W$ on both sides and applying property $(d)$ of Proposition \ref{prop:SIproperties} gives
$$E[X_t\,|\, \mathcal{F}_s^W]=X_s+E\left[\int_s^t g_u\, dW_u\,\middle|\,\mathcal{F}_s^W\right]=X_s$$
which satisfies condition (3) of Definition \ref{def:martingale}. Adaptiveness and integrability follow directly from $g\in\mathcal{L}^2$, and hence $X_t$ is a $\{\mathcal{F}_t^W\}$-martingale.
\end{proof}

Any stochastic integral $\int_0^tg_sdW_s$ with $g\in\mathcal{L}^2$ is therefore a martingale. Recalling the SDE $\eqref{eq:SDE}$, this means that when the drift coefficient $\mu(t,X_t)$ vanishes, the process $X_t$ itself becomes a martingale.



\subsection{Itô's Formula}

Itô's formula is a central result of stochastic calculus, providing the stochastic differential of any smooth transformation $f(t,X_t)$ of a diffusion process. To state it in full generality, we consider a $n$-dimensional process $X=(X^1,...,X^n)^\top$ driven by $d$ Brownian motions $W=(W^1,...,W^d)^\top$ with instantaneous correlation structure
\begin{equation}
    dW^i_t \cdot dW^j_t = \rho_{ij}\,dt,
\end{equation}
where $\rho$ is the correlation matrix.

The dynamics of $X$ are given by
\begin{equation}
    d\underbrace{\begin{bmatrix}
    X_t^1\\ \vdots\\ X_t^n
    \end{bmatrix}}_{n\times 1}=
    \underbrace{\begin{bmatrix}
    \mu_t^1\\ \vdots\\ \mu_t^n
    \end{bmatrix}}_{n\times 1}dt+
    \underbrace{\begin{bmatrix}
    \sigma_t^{11} & \sigma_t^{12} & \dots & \sigma_t^{1d}\\
    \sigma_t^{21} & \sigma_t^{22} & \dots & \sigma_t^{2d}\\
    \vdots & \vdots & \ddots & \vdots\\
    \sigma_t^{n1} & \sigma_t^{n2} & \dots & \sigma_t^{nd}\\
    \end{bmatrix}}_{n\times d}
    d\underbrace{\begin{bmatrix}
    W^1_t\\ \vdots\\ W^d_t
    \end{bmatrix}}_{d\times 1}
\label{eq:multiSDE}
\end{equation}
which is a multidimensional generalization of \eqref{eq:SDE}, written compactly as $dX_t=\mu_tdt+\sigma_tdW_t$, where each component satisfies
$$dX_t^i=\mu^i_tdt+\sum_{j=1}^d\sigma^{ij}_tdW_t^j.$$
For notational convenience, the dependence of the drift $\mu_t\in\mathbb{R}^n$ and the diffusion matrix $\sigma_t\in\mathbb{R}^{n\times d}$ on the current state $X_t$ is suppressed throughout, with the understanding that $\mu_t=\mu_t(X_t)$ and $\sigma_t=\sigma_t(X_t)$ as in \eqref{eq:SDE}.



\begin{theorem}[Itô's Formula]
Let the $n$-dimensional process $X$ have dynamics given by \eqref{eq:multiSDE} and let $f \in C^{1,2}(\mathbb{R}_+ \times \mathbb{R}^n)$. Then $f(t,X_t)$ has a stochastic differential given by
\begin{equation}
    df(t,X_t) = \left\{
        \frac{\partial f}{\partial t}
        + \sum_{i=1}^n \mu_t^i \frac{\partial f}{\partial x^i}
        + \frac{1}{2}\,\mathrm{tr}\!\left[\sigma_t^\top H_f \sigma_t\right]
    \right\}dt
    + \sum_{i=1}^n \frac{\partial f}{\partial x^i}\,\sigma_t^i\,dW_t,
    \label{eq:Ito}
\end{equation}
where $H$ denotes the Hessian matrix with
$$H_{ij}=\frac{\partial^2f}{\partial {x^i}\partial {x^j}},$$
and $\text{tr}$ denotes the trace of a matrix, which is defined as $\text{tr}(A)=\sum_i A_{ii}$ for any square matrix $A$.
\label{thm:ito}
\end{theorem}

\begin{remark}[Itô Multiplication Table]
Theorem~\ref{thm:ito} follows from the formal multiplication rules
\begin{equation}
\begin{cases}
    (dt)^2 = 0,\\
    dt \cdot dW^i_t = 0, \quad i=1,\ldots,d\\
    (dW^i_t)^2 = dt, \quad i=1,\ldots,d\\
    dW^i_t \cdot dW^j_t = \rho_{ij}\,dt,
\end{cases}
\end{equation}
where the last rule reduces to $dW^i_t \cdot dW^j_t = 0$ for $i \neq j$ 
in the independent case.
\label{rem:multitable}
\end{remark}




\begin{remark}[Infinitesimal Generator]
\label{rem:generator}
The drift term in \eqref{eq:Ito} decomposes naturally into an explicit 
time derivative $\partial_t f$ and a purely spatial part determined by 
$\mu_t$ and $\sigma_t$. Defining the latter as the \emph{infinitesimal generator}
\begin{equation}
    \mathcal{A}f(x) = \sum_{i=1}^n \mu_t^i \frac{\partial f}{\partial x^i}
    + \frac{1}{2}\operatorname{tr}\!\left[\sigma_t^\top H_f \sigma_t\right],
    \label{eq:generator}
\end{equation}
where $H_f$ denotes the Hessian matrix with entries 
$[H_f]_{ij} = \frac{\partial^2 f}{\partial x^i \partial x^j}$, 
Itô's formula takes the compact form
\begin{equation}
    df = \left(\partial_t f + \mathcal{A}f\right)dt 
    + \sum_{i=1}^n \frac{\partial f}{\partial x^i}\sigma_t^i\, dW_t.
\end{equation}
In particular, $f(t, X_t)$ is a martingale if and only if 
$\partial_t f + \mathcal{A}f = 0$, a partial differential 
equation in $(t,x)$ whose coefficients are exactly those of the 
SDE \eqref{eq:multiSDE}. This connection between the dynamics of $f(t,X_t)$ and the theory of partial differential equations is made precise by the Feynman--Kac formula of Section~\ref{subsec:fk}, and underlies the pricing framework developed in later sections.
\end{remark}


\subsection{Existence and Uniqueness of Solutions}

We now establish conditions under which the SDE admits a unique solution. More precisely, we seek a process $X$ satisfying \eqref{eq:SDE} for all $t\geq0$.

\begin{proposition}
\label{prop:sde_existence}
    Suppose that there exists a constant K such that the following conditions are satisfied for all $x,y$ and $t$.
    \begin{align}
     ||\mu_t(x)-\mu_t(y)||&\leq K ||x - y||\\
     ||\sigma_t(x)-\sigma_t(y)||&\leq K ||x - y||\\
     ||\mu_t(x)||+||\sigma_t(x)||&\leq K (1+||x||)
    \end{align}
     Then the SDE \eqref{eq:SDE} admits a unique solution. The solution has the properties
    \begin{enumerate}
        \item $X$ is $\mathcal{F}_t^W$-adapted.
        \item $X$ has continuous trajectories.
        \item $X$ is a Markov process.
    \end{enumerate}
\end{proposition}

The first two conditions of Proposition \ref{prop:sde_existence} are \textit{Lipschitz conditions} on $\mu$ and $\sigma$ respectively, ensuring the coefficients do not vary too \textit{wildly} in space. The third is a \textit{linear growth condition} preventing the solution from blowing up in finite time. Notably, the solution inherits the Markov property. 


\begin{theorem}[Markov Property - {\citet[p.~64]{Filipovic2009}} ]
\label{thm:markov}
    Suppose $\mu_t(x)$ and $\sigma_t(x)$ are continuous in $(t,x)$, and assume 
    that for every initial point $(t_0, x_0) \in \mathbb{R}_+ \times \mathbb{R}^n$, 
    there exists a unique solution $X = X^{(t_0,x_0)}$ of \eqref{eq:SDE}. 
    Then $X$ has the Markov property. That is, for every bounded measurable 
    function $f$ on $\mathbb{R}^n$, there exists a measurable function $F$ on 
    $\mathbb{R}_+ \times \mathbb{R}_+ \times \mathbb{R}^n$ such that
    \begin{equation}
        \mathbb{E}[f(X_T) \mid \mathcal{F}_t] = F(t, T, X_t), \quad t \leq T.
    \end{equation}
\end{theorem}

This property of diffusion processes is important to the pricing framework 
developed later, as it ensures that $\mathbb{E}[f(X_T) \mid \mathcal{F}_t]$ 
depends only on the current state $X_t$ rather than on the full history of 
the process.



\subsection{The Feynman--Kac Formula}
\label{subsec:fk}

In pricing theory, asset prices are expressed as conditional expectations 
of the form $\mathbb{E}[f(X_T) \mid X_t = x]$ under a pricing measure. 
The Feynman--Kac formula establishes that these expectations solve a parabolic 
PDE, so that prices can be computed either by simulating the underlying 
diffusion or by solving the corresponding PDE.

\begin{theorem}[Feynman--Kac, {\citealp[Prop.~5.6]{Bjork2020}}]
\label{thm:fk}
Let $\varphi : \mathbb{R}^n \to \mathbb{R}$ be continuous and satisfy a polynomial growth condition, let $k : [0,T] \times \mathbb{R}^n \to \mathbb{R}$ be bounded and continuous, and suppose that
\[
f \in C([0,T]\times\mathbb{R}^n)\cap C^{1,2}([0,T)\times\mathbb{R}^n)
\]
solves the terminal-value problem
\begin{equation}
\begin{cases}
    \partial_t f(t,x)+\mathcal{A}f(t,x)-k(t,x)\,f(t,x)=0,
    & (t,x)\in[0,T)\times\mathbb{R}^n, \\[4pt]
    f(T,x)=\varphi(x),
    & x\in\mathbb{R}^n,
\end{cases}
\label{eq:fk_pde}
\end{equation}
where $\mathcal{A}$ is the infinitesimal generator defined in \eqref{eq:generator}. Then $f$ admits the probabilistic representation
\begin{equation}
f(t,x)
=
\mathbb{E}\!\left[
    \exp\!\left(-\int_t^T k(s,X_s)\,ds\right)\varphi(X_T)
    \,\middle|\,
    X_t=x
\right],
\label{eq:fk_formula}
\end{equation}
where $X$ is the diffusion process with dynamics \eqref{eq:multiSDE}.
\end{theorem}

\begin{proof}
See \citet[Prop.~5.6]{Bjork2020}. A version under weaker regularity conditions is given in \citet[Thm.~5.3]{Friedman1975}.
\end{proof}

Two aspects of \eqref{eq:fk_formula} are particularly important. First, the representation depends on the past only through the current state $(t,x)$, reflecting the Markov property from Theorem~\ref{thm:markov}. Second, if $k\equiv 0$, then \eqref{eq:fk_formula} reduces to
\[
f(t,x)=\mathbb{E}\!\left[\varphi(X_T)\mid X_t=x\right],
\]
while \eqref{eq:fk_pde} reduces to $\partial_t f+\mathcal{A}f=0$, exactly the drift condition from Remark~\ref{rem:generator}. This result will be central in the pricing theory developed below, where the discount factor is generated by the short rate and the Feynman--Kac formula yields the PDE representation of arbitrage-free bond prices.






























\section{Arbitragefree Pricing}

We now apply the stochastic calculus developed above to model a financial market, with the goal of pricing contingent claims in an arbitrage-free manner.




The market is modeled on a filtered probability space $(\Omega, \mathcal{F}, 
\{\mathcal{F}_t\}_{t \geq 0}, \mathbb{P})$ supporting a $d$-dimensional 
standard Brownian motion $W = (W^1, \ldots, W^d)^\top$ with independent 
components. It consists of $N$ risky assets $S^1, \ldots, S^N$ and a 
risk-free bank account $B$, whose dynamics under $\mathbb{P}$ are given by
\begin{equation}
    \begin{cases}
        dS_t^i = \mu_t^i \, dt + \displaystyle\sum_{j=1}^{d} 
        \sigma_t^{ij} \, dW_t^j, & i = 1, \ldots, N, \\[8pt]
        dB_t = r_t B_t \, dt, & B_0 = 1,
    \end{cases}
    \label{eq:market_dynamics}
\end{equation}
where $\mu^i$, $\sigma^{ij}$, and $r$ are $\{\mathcal{F}_t\}$-adapted processes, with dependence on the current state $S_t$ suppressed as in \eqref{eq:SDE}. We interpret $\mu^i_t$ as the \emph{local mean rate of return}, $\sigma^{ij}_t$ as the \emph{volatility} of asset $S^i$, and $r_t$ as the short rate. The bank account has solution $B_t = \exp\!\left(\int_0^t r_s\, ds\right)$ and serves as the numeraire. The conditions under which this is arbitrage-free are characterized 
by the following theorem.

% I starten af kapitel 11, fortæller forfatteren hvordan man kan læse kapitlet. Ifølge denne paragraf, kan man erstatte udtrykket NFLVR med "No Arbitrage" og ignorere betingelsen "locally bounded", da dette er altid opfyldt for wiener drevne modeller.

\begin{theorem}[The First Fundamental Theorem]
    Consider the market model $B,S^1,...,S^N$ where we assume that $B>0$, \textcolor{red}{$P$-a.s.} for all $t\geq0$. Then the following conditions are equivalent:
    \begin{itemize}
        \item The model is arbitragefree
        \item There exists a measure $\mathbb{Q}\sim \mathbb{P}$ such that the processes $\frac{B}{B},\frac{S^1}{B},...,\frac{S^N}{B},$
        are martingales under $\mathbb{Q}$.
    \end{itemize}
\end{theorem}


With the arbitrage-free condition established, we turn to the pricing of 
contingent claims. Given a $T$-claim $X$, the First Fundamental Theorem 
guarantees the existence of an equivalent martingale measure $\mathbb{Q}$ under which yields the following pricing formula.

\begin{theorem}[Risk-Neutral Valuation Formula]
\label{thm:riskneutral}
Let the market be arbitrage-free and let $\mathbb{Q}$ be an equivalent 
martingale measure associated with the bank account numeraire $B_t$. 
Then the arbitrage-free price process of a $T$-claim $X$ is given by
\begin{equation}
    \Pi_t[X] = \mathbb{E}^{\mathbb{Q}}\!\left[
    e^{-\int_t^T r_s\, ds} X \,\bigg|\, \mathcal{F}_t \right]
    = \mathbb{E}^{\mathbb{Q}}\!\left[
    \frac{B_t}{B_T} X \,\bigg|\, \mathcal{F}_t \right].
    \label{eq:riskneutral}
\end{equation}
\end{theorem}

\begin{example}
\label{ex: ZCB price}
As a standard example, the price of a contract paying one unit of currency at time $T$, being the trivial claim $X \equiv 1$, is immediately given by
Theorem \ref{thm:riskneutral} as
\begin{equation}
    P(t,T) = \mathbb{E}^{\mathbb{Q}}\!\left[
    e^{-\int_t^T r_s\, ds} \,\bigg|\, \mathcal{F}_t \right],
    \label{eq:zcb}
\end{equation}
with $P(T,T) = 1$.    
\end{example}




We close this section by remarking that $\mathbb{Q}$ need not be unique. By the Second Fundamental 
Theorem, uniqueness holds if and only if the market is complete. As we do not impose completeness, $\mathbb{Q}$ is not unique. For the remainder of the thesis we therefore fix a $\mathbb{Q}$, acknowledging that the 
resulting dynamics may not be the unique arbitrage-free specification. 
Under this $\mathbb{Q}$, the market dynamics are given by
\begin{equation}
    \begin{cases}
        dS_t^i = r_t \, dt + \displaystyle\sum_{j=1}^{d} 
        \sigma_t^{ij} \, dW_t^{\mathbb{Q},j}, & i = 1, \ldots, N, \\[8pt]
        dB_t = r_t B_t \, dt, & B_0 = 1. 
    \end{cases}
    \label{eq:Q_dynamics}
\end{equation}



\section{Interest Rate Theory}

We now introduce the interest rate instruments to be modeled in this thesis, in line with \citet{Filipovic2009}. The contract whose price was introduced in Example \ref{ex: ZCB price} is known as a \textit{zero-coupon bond}, which we now define formally. 

\begin{definition}[\textit{Zero Coupon Bond}]
A zero coupon bond with maturity $T$, also called a $T$-bond, is a contract guaranteeing the holder one unit of currency at time $T$. The price at time $t\leq T$ is denoted $P(t,T)$, with terminal condition
$$P(T,T)=1.$$
\end{definition}

We assume there exists a frictionless market for $T$-bonds for every maturity $T>0$. As shown in Example \ref{ex: ZCB price}, the no-arbitrage price of a $T$-bond at time $t\leq T$ is given by 
\begin{equation}
    P(t,T)=E^Q\left[e^{-\int_t^T r_s\,ds}\,\middle|\,\mathcal{F}_t\right]
    \label{ZCBprice}
\end{equation}

The mapping $T \mapsto P(t,T)$ is known as the term structure of interest rates. From it, we derive three related quantities.


\begin{definition}[\textit{Continuously Compounded Spot Rate}]
    The continuously compounded spot rate $y(t,T)$ of a $T$-bond at time $t$ is defined by 
    \begin{equation}
        P(t,T)=e^{-y(t,T)(T-t)} 
        \label{eq:yield1}
    \end{equation}
    or equivalently
    \begin{equation}
        y(t,T)=-\frac{\log P(t,T)}{T-t} 
        \label{eq:yield2}
    \end{equation}
\end{definition}

\begin{definition}[\textit{Instantaneous Forward Rate}]
The instantaneous forward rate for maturity $T$ observed at time $t$ is defined by
\begin{equation}
f(t,T)=-\frac{\partial}{\partial T}\log P(t,T),
\label{eq:inst_forward}
\end{equation}
provided the derivative exists. 
\end{definition}

The function $T\mapsto f(t,T)$ is called the forward curve. The instantaneous short rate arises as the limit as maturity approaches $t$.

\begin{definition}[Instantaneous Short Rate]
The instantaneous short rate at time $t$ is defined by
\begin{equation}
    r_t=\lim_{T\to t} f(t,T).
    \label{eq:instshortrate}
\end{equation}
\end{definition}


\begin{definition}[Simply Compounded Forward Rate]
The simply compounded forward rate $F(t;T_{i-1},T_i)$ over the accrual period $[T_{i-1},T_i]$, is defined by
\begin{equation}
1+\Delta_i F(t;T_{i-1},T_i)=\frac{P(t,T_{i-1})}{P(t,T_i)},
\label{eq:discrete_forward}
\end{equation}
where $\Delta_i=T_i-T_{i-1}$ denotes the accrual fraction.
\end{definition}

A classic market example is LIBOR (London Interbank Offered Rate), with rate $L(t;T_{i-1},T_i)$. In practice, LIBOR may embed liquidity and credit risk, so the identification $L=F$ need not hold \citep[p. 7]{Filipovic2009}. We therefore adopt the following assumption.



\begin{assumption}[\textit{Single Curve}]
A single discount curve, determined entirely by the bond prices $\{P(t,T)\}_{T\geq t}$, is used for both discounting and forwarding. 
\label{as:singlecurve}
\end{assumption}

Under Assumption \ref{as:singlecurve} the family $\{P(t,T)\}_{T\geq t}$ fully encodes the term structure, and $y(t,T)$, $f(t,T)$ and $F(t;T_{i-1},T_i)$ may all be derived from it. 















\subsection{Interest Rate Swaps}

An interest rate swap is an agreement to exchange a fixed payment stream for a floating payment stream \citep{Filipovic2009}. Consider the standard fixed-for-floating interest rate swap initiated at time $t$ and maturing at $T=t+\tau$, where $\tau$ is time to maturity. Let
\[
T_0=t<T_1<\dots<T_n=T
\]
denote the payment dates and define the accrual fractions
\[
\Delta_i = T_i-T_{i-1}, \qquad i=1,\dots,n.
\]

Let $N$ denote the notional. For fixed rate $S(t,\tau)$, the value of the fixed leg at time $t$ is
\begin{equation}
V_{\mathrm{fix}}(t)
=
N S(t,\tau)
\sum_{i=1}^{n}\Delta_i P(t,T_i).
\label{eq:fixed_leg}
\end{equation}

Under Assumption \ref{as:singlecurve}, the floating rate over $[T_{i-1},T_i]$ satisfies
\[
1+\Delta_i F(t;T_{i-1},T_i)
=
\frac{P(t,T_{i-1})}{P(t,T_i)}.
\]
Using \eqref{eq:discrete_forward}, the floating rate over $[T_{i-1},T_i]$ satisfies
\[
N\cdot\Delta_i F(t;T_{i-1},T_i)\cdot P(t,T_i) = N\bigl(P(t,T_{i-1})-P(t,T_i)\bigr).
\]
Summing over all payment dates, the value of the floating leg is
\begin{equation}
V_{\mathrm{float}}(t) = N\sum_{i=1}^{n}\bigl(P(t,T_{i-1})-P(t,T_i)\bigr) = N\bigl(1-P(t,T)\bigr),
\label{eq:floating_leg}
\end{equation}
where the sum telescopes and $P(t,t)=1$ for a spot-starting swap.

The par swap rate is the fixed rate that sets the value to zero. Solving
\(
V_{\mathrm{float}}(t)-V_{\mathrm{fix}}(t)=0
\)
yields
\begin{equation}
S(t,\tau)
=
\frac{1-P(t,T)}
{\sum_{i=1}^{n} \Delta_i P(t,T_i)}.
\label{eq:swap_rate}
\end{equation}

Hence, under the single-curve assumption, par swap rates are fully determined by the discount curve through \eqref{eq:swap_rate}.
