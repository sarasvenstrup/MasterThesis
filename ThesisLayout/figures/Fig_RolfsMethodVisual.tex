\begin{tikzpicture}[
    scale=0.7,
    transform shape,
    neuron/.style={circle, minimum size=10mm, inner sep=0pt},
    >=stealth
]

\definecolor{darkgrey}{RGB}{90,90,90}

% -------- Step bracket tuning --------
\def\StepTick{4pt}
\def\StepRise{10pt}
\def\StepGap{8pt}
\def\TickSkew{6pt}

% ======================
% Input layer (8 nodes)
% ======================
\foreach \i in {1,...,8} {
    \node[neuron, fill=grey!40] (S\i) at (0, {(4.5-\i)*1.2}) {$S_{\i}$};
}

% ======================
% Latent layer
% ======================
\node[neuron, fill=red!70] (Ztop) at (4,  1.9) {};
\node[above=2mm] at (Ztop.north) {$\tau_i = 1,\ldots,30$};

\node[neuron, fill=red!70] (Z1)  at (4,  0.4) {$z_{1}$};
\node                    (Zvd) at (4, -0.6) {$\vdots$};
\node[neuron, fill=red!70] (Zd)  at (4, -1.8) {$z_{l}$};

% ======================
% Hidden layer (10 nodes)
% ======================
\foreach \i in {1,...,10} {
    \node[neuron, fill=red!40] (H\i) at (8, {(5.5-\i)*1.15}) {$h_{\i}$};
}

% ======================
% Output layer (3 nodes)
% ======================
\node[neuron, fill=red!20] (P1)  at (12,  2.5) {$P_{\tau_1}$};
\node[neuron, fill=red!20] (P2)  at (12,  1.0) {$P_{\tau_2}$};
\node at (12, -0.2) {$\vdots$};
\node[neuron, fill=red!20] (P30) at (12, -2.2) {$P_{\tau_{30}}$};

% ======================
% Reconstructed swaps
% ======================
\node[neuron, fill=grey!10] (St1) at (16,  2.5) {$\tilde{S}_{1}$};
\node[neuron, fill=grey!10] (St2) at (16,  1.0) {$\tilde{S}_{2}$};
\node at (16, -0.2) {$\vdots$};
\node[neuron, fill=grey!10] (St8) at (16, -2.2) {$\tilde{S}_{8}$};

% ======================
% Connections
% ======================
\foreach \i in {1,...,8} {
    \draw[->, draw=darkgrey] (S\i) -- (Z1);
    \draw[->, draw=darkgrey] (S\i) -- (Zd);
}

\foreach \j in {1,...,10} {
    \draw[->, draw=darkgrey] (Ztop) -- (H\j);
    \draw[->, draw=darkgrey] (Z1)   -- (H\j);
    \draw[->, draw=darkgrey] (Zd)   -- (H\j);
}

\foreach \i in {1,...,10} {
    \draw[->, draw=darkgrey]         (H\i) -- (P1);
    \draw[->, dashed, draw=darkgrey] (H\i) -- (P2);
    \draw[->, dotted, draw=darkgrey] (H\i) -- (P30);
}

\draw[->, draw=darkgrey] (P1) -- (St1);
\draw[->, draw=darkgrey] (P1) -- (St2);
\draw[->, draw=darkgrey] (P2) -- (St2);
\draw[->, draw=darkgrey] (P1)  -- (St8);
\draw[->, draw=darkgrey] (P2)  -- (St8);
\draw[->, draw=darkgrey] (P30) -- (St8);

% ======================
% Step brackets
% ======================

\coordinate (StepY) at ([yshift=\StepRise]H1.north);

\newcommand{\StepBracket}[3]{%
  \draw[thick] (#1) -- ++(0,-\StepTick);
  \draw[thick] (#1) -- (#2);
  \draw[thick] (#2) -- ++(0,-\StepTick);
  \node[above=4pt] at ($(#1)!0.5!(#2)$) {\textbf{#3}};
}

% Step 1
\coordinate (B0) at ([xshift=\TickSkew]S1.center |- StepY);
\coordinate (B1) at ([xshift=-\TickSkew]Ztop.center |- StepY);
\coordinate (S1L) at (B0);
\coordinate (S1R) at ([xshift=-\StepGap]B1);
\StepBracket{S1L}{S1R}{Step 1}

% Step 2
\coordinate (B2) at ([xshift=\TickSkew]Ztop.center |- StepY);
\coordinate (B3) at ([xshift=-\TickSkew]H1.center |- StepY);
\coordinate (S2L) at ([xshift=\StepGap]B2);
\coordinate (S2R) at ([xshift=-\StepGap]B3);
\StepBracket{S2L}{S2R}{Step 2}

% Step 3
\coordinate (B4) at ([xshift=\TickSkew]H1.center |- StepY);
\coordinate (B5) at ([xshift=-\TickSkew]P1.center |- StepY);
\coordinate (S3L) at ([xshift=\StepGap]B4);
\coordinate (S3R) at ([xshift=-\StepGap]B5);
\StepBracket{S3L}{S3R}{Step 3}

% Step 4
\coordinate (B6) at ([xshift=\TickSkew]P1.center |- StepY);
\coordinate (B7) at ([xshift=-\TickSkew]St1.center |- StepY);
\coordinate (S4L) at ([xshift=\StepGap]B6);
\coordinate (S4R) at (B7);
\StepBracket{S4L}{S4R}{Step 4}
