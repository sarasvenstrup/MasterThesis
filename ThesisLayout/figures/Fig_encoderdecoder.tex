    \foreach \t in {D1,D2,D3} {

\caption{Undercomplete autoencoder with a symmetric encoder--decoder
architecture. The encoder $E$ compresses the input $x \in \mathbb{R}^d$

decoder $D$ reconstructs $\hat{x} \in \mathbb{R}^d$ from $z$.}
